Example use case · उदाहरण

Fire alarm beep — Sound buzzer when digital output detects flame.

const int fireDO = 3;

void setup() {
  pinMode(fireDO, INPUT);
  pinMode(5, OUTPUT);
}

void loop() {
  if (digitalRead(fireDO) == LOW) {
    tone(5, 1500, 300);
  }
  delay(100);
}
